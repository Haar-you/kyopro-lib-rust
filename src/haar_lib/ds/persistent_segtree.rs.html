3use std::cell::RefCell;
4use std::ops::RangeBounds;
5use std::rc::Rc;

7use crate::algebra::traits::Monoid;

10#[derive(Clone, Debug)]
11struct Node<T> {
12    value: T,
13    left: Option<Rc<RefCell<Node<T>>>>,
14    right: Option<Rc<RefCell<Node<T>>>>,
15}
16
17impl<T> Node<T> {
18    fn new(value: T) -> Self {
19        Self {
20            value,
21            left: None,
22            right: None,
23        }
24    }
25}
26
27/// 永続セグメントツリー
28#[derive(Clone, Debug)]
29pub struct PersistentSegtree<M: Monoid> {
30    root: Option<Rc<RefCell<Node<M::Element>>>>,
31    monoid: M,
32    to: usize,
33    original_size: usize,
34}
35
36impl<M: Monoid + Clone> PersistentSegtree<M>
37where
38    M::Element: Clone,
39{
40    /// 長さ`n`の[`PersistentSegtree`]を生成する。
41    pub fn new(n: usize, monoid: M) -> Self {
42        let seq = vec![monoid.id(); n];
43        Self::from_vec(seq, monoid)
44    }
45
46    /// [`Vec`]から[`PersistentSegtree`]を構築する。
47    pub fn from_vec(a: Vec<M::Element>, monoid: M) -> Self {
48        let n = a.len();
49        let to = n.next_power_of_two();
50        let root = Some(Self::__init(0, to, &a, &monoid));
51        Self {
52            root,
53            monoid,
54            to,
55            original_size: n,
56        }
57    }
58
59    fn __init(
60        from: usize,
61        to: usize,
62        seq: &[M::Element],
63        monoid: &M,
64    ) -> Rc<RefCell<Node<M::Element>>> {
65        if to - from == 1 {
66            Rc::new(RefCell::new(Node::new(seq[from].clone())))
67        } else {
68            let mid = (from + to) / 2;
69            let mut node = Node::new(monoid.id());
70
71            let lv = if seq.len() > from {
72                let left = Self::__init(from, mid, seq, monoid);
73                let lv = left.borrow().value.clone();
74                node.left = Some(left);
75                lv
76            } else {
77                monoid.id()
78            };
79
80            let rv = if seq.len() > mid {
81                let right = Self::__init(mid, to, seq, monoid);
82                let rv = right.borrow().value.clone();
83                node.right = Some(right);
84                rv
85            } else {
86                monoid.id()
87            };
88
89            node.value = monoid.op(lv, rv);
90
91            Rc::new(RefCell::new(node))
92        }
93    }
94
95    fn __set(
96        node: Rc<RefCell<Node<M::Element>>>,
97        from: usize,
98        to: usize,
99        pos: usize,
100        value: &M::Element,
101        monoid: &M,
102    ) -> Rc<RefCell<Node<M::Element>>> {
103        if to <= pos || pos < from {
104            node
105        } else if pos <= from && to <= pos + 1 {
106            Rc::new(RefCell::new(Node::new(value.clone())))
107        } else {
108            let mid = (from + to) / 2;
109
110            let lp = node
111                .borrow()
112                .left
113                .clone()
114                .map(|left| Self::__set(left, from, mid, pos, value, monoid));
115            let rp = node
116                .borrow()
117                .right
118                .clone()
119                .map(|right| Self::__set(right, mid, to, pos, value, monoid));
120
121            let mut s = Node::new(
122                monoid.op(
123                    lp.as_ref()
124                        .map_or(monoid.id(), |l| l.borrow().value.clone()),
125                    rp.as_ref()
126                        .map_or(monoid.id(), |r| r.borrow().value.clone()),
127                ),
128            );
129
130            s.left = lp;
131            s.right = rp;
132
133            Rc::new(RefCell::new(s))
134        }
135    }
136
137    /// `i`番目の要素を`value`にする。
138    pub fn assign(&self, i: usize, value: M::Element) -> Self {
139        let new_root = Self::__set(
140            self.root.clone().unwrap(),
141            0,
142            self.to,
143            i,
144            &value,
145            &self.monoid,
146        );
147
148        Self {
149            root: Some(new_root),
150            monoid: self.monoid.clone(),
151            to: self.to,
152            original_size: self.original_size,
153        }
154    }
155
156    fn __fold(
157        node: Rc<RefCell<Node<M::Element>>>,
158        from: usize,
159        to: usize,
160        l: usize,
161        r: usize,
162        monoid: &M,
163    ) -> M::Element {
164        if l <= from && to <= r {
165            node.borrow().value.clone()
166        } else if to <= l || r <= from {
167            monoid.id()
168        } else {
169            let mid = (from + to) / 2;
170
171            let lv = node.borrow().left.clone().map_or(monoid.id(), |left| {
172                Self::__fold(left, from, mid, l, r, monoid)
173            });
174
175            let rv = node.borrow().right.clone().map_or(monoid.id(), |right| {
176                Self::__fold(right, mid, to, l, r, monoid)
177            });
178
179            monoid.op(lv, rv)
180        }
181    }
182
183    /// 範囲`range`で計算を集約して返す。
184    pub fn fold(&self, range: impl RangeBounds<usize>) -> M::Element {
185        let (start, end) = range_bounds_to_range(range, 0, self.original_size);
186        Self::__fold(
187            self.root.clone().unwrap(),
188            0,
189            self.to,
190            start,
191            end,
192            &self.monoid,
193        )
194    }
195}
