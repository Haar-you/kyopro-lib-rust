1//! 動的遅延セグメント木
2use crate::algebra::action::Action;
3use std::ops::Range;
4use std::ptr;
5
6#[derive(Clone, Debug)]
7struct Node<A: Action> {
8    value: A::Output,
9    lazy: A::Lazy,
10    left: *mut Node<A>,
11    right: *mut Node<A>,
12}
13
14impl<A: Action> Node<A> {
15    fn new(action: A) -> Self {
16        Self {
17            value: action.fold_id(),
18            lazy: action.update_id(),
19            left: ptr::null_mut(),
20            right: ptr::null_mut(),
21        }
22    }
23}
24
25/// 動的遅延セグメント木
26#[derive(Clone, Debug)]
27pub struct DynamicLazySegtree<A: Action> {
28    root: *mut Node<A>,
29    action: A,
30    to: usize,
31}
32
33impl<A: Action + Copy> DynamicLazySegtree<A>
34where
35    A::Output: Clone + PartialEq,
36    A::Lazy: Clone + PartialEq,
37{
38    /// `DynamicLazySegtree<A>`を生成する。
39    pub fn new(action: A) -> Self {
40        Self {
41            root: Box::into_raw(Box::new(Node::new(action))),
42            action,
43            to: 1,
44        }
45    }
46
47    fn _propagate(&self, t: *mut Node<A>, from: usize, to: usize) {
48        assert!(!t.is_null());
49        let lazy = unsafe { (*t).lazy.clone() };
50
51        if lazy == self.action.update_id() {
52            return;
53        }
54        if to - from > 1 {
55            unsafe {
56                if (*t).left.is_null() {
57                    (*t).left = Box::into_raw(Box::new(Node::new(self.action)));
58                }
59                let left = (*t).left;
60                (*left).lazy = self.action.update(lazy.clone(), (*left).lazy.clone());
61
62                if (*t).right.is_null() {
63                    (*t).right = Box::into_raw(Box::new(Node::new(self.action)));
64                }
65                let right = (*t).right;
66                (*right).lazy = self.action.update(lazy.clone(), (*right).lazy.clone());
67            }
68        }
69        let len = to - from;
70        unsafe {
71            (*t).value = self.action.convert((*t).value.clone(), lazy, len);
72            (*t).lazy = self.action.update_id();
73        }
74    }
75
76    fn _update(
77        &self,
78        mut cur: *mut Node<A>,
79        from: usize,
80        to: usize,
81        s: usize,
82        t: usize,
83        value: A::Lazy,
84    ) -> *mut Node<A> {
85        if cur.is_null() {
86            cur = Box::into_raw(Box::new(Node::new(self.action)));
87        }
88
89        self._propagate(cur, from, to);
90
91        if to - from == 1 {
92            if s <= from && to <= t {
93                unsafe {
94                    (*cur).lazy = self.action.update(value, (*cur).lazy.clone());
95                }
96            }
97            self._propagate(cur, from, to);
98            return cur;
99        }
100
101        if to < s || t < from {
102            return cur;
103        }
104        if s <= from && to <= t {
105            unsafe {
106                (*cur).lazy = self.action.update(value, (*cur).lazy.clone());
107            }
108            self._propagate(cur, from, to);
109            return cur;
110        }
111
112        let mid = (from + to) / 2;
113        unsafe {
114            (*cur).left = self._update((*cur).left, from, mid, s, t, value.clone());
115            (*cur).right = self._update((*cur).right, mid, to, s, t, value);
116            (*cur).value = self
117                .action
118                .fold((*(*cur).left).value.clone(), (*(*cur).right).value.clone());
119        }
120        cur
121    }
122
123    /// 範囲`s..t`を`value`で更新する。
124    pub fn update(&mut self, Range { start: s, end: t }: Range<usize>, value: A::Lazy) {
125        loop {
126            if t <= self.to {
127                break;
128            }
129            self.to *= 2;
130
131            let mut new_root = Box::new(Node::new(self.action));
132            new_root.left = self.root;
133
134            self.root = Box::into_raw(new_root);
135        }
136
137        self._update(self.root, 0, self.to, s, t, value);
138    }
139
140    fn _fold(&self, cur: *mut Node<A>, from: usize, to: usize, s: usize, t: usize) -> A::Output {
141        if cur.is_null() {
142            return self.action.fold_id();
143        }
144
145        self._propagate(cur, from, to);
146        if to <= s || t <= from {
147            return self.action.fold_id();
148        }
149        if s <= from && to <= t {
150            return unsafe { (*cur).value.clone() };
151        }
152
153        let mid = (from + to) / 2;
154        let lv = self._fold(unsafe { (*cur).left }, from, mid, s, t);
155        let rv = self._fold(unsafe { (*cur).right }, mid, to, s, t);
156
157        self.action.fold(lv, rv)
158    }
159
160    /// 範囲`s..t`で計算を集約する。
161    pub fn fold(&mut self, Range { start: s, end: t }: Range<usize>) -> A::Output {
162        self._fold(self.root, 0, self.to, s, t)
163    }
164}
