10/// Z algorithm
11///
12/// $k \in [1, n]$について、
13/// $s = s_1 s_2 \ldots s_n$と$s$の$k$文字目以降$s_k s_{k+1} \ldots s_n$の最長共通接頭辞の長さを求める。
14pub fn zalgo<T: PartialEq>(s: &[T]) -> Vec<usize> {
15    let n = s.len();
16    let mut ret = vec![0; n];
17    let mut j = 0;
18
19    for i in 1..n {
20        if i + ret[i - j] < j + ret[j] {
21            ret[i] = ret[i - j];
22        } else {
23            let mut k = if j + ret[j] >= i { j + ret[j] - i } else { 0 };
24            while i + k < n && s[k] == s[i + k] {
25                k += 1;
26            }
27            ret[i] = k;
28            j = i;
29        }
30    }
31
32    ret[0] = n;
33
34    ret
35}
