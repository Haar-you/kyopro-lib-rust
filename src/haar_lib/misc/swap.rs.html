1//! 二次元`Vec`の要素swap
2
3/// `a[i1][j1]`と`a[i2][j2]`を交換する。
4pub fn swap_vv<T>(a: &mut [Vec<T>], i1: usize, j1: usize, i2: usize, j2: usize) {
5    let p1: *mut T = &mut a[i1][j1];
6    let p2: *mut T = &mut a[i2][j2];
7
8    unsafe {
9        p1.swap(p2);
10    }
11}
