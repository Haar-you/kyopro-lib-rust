7/// 素数$P$上の数論変換 (Number Theoretic Transform)
8///
9/// `PRIM_ROOT`は`P`の原始根。
10#[derive(Clone)]
11pub struct NTT<P: PrimeMod> {
12    _phantom: PhantomData<P>,
13}
14
15impl<P: PrimeMod> NTT<P> {
16    const MAX_POWER: usize = (P::PRIME_NUM as usize - 1).trailing_zeros() as usize;
17    const MAX_SIZE: usize = 1 << Self::MAX_POWER;
18    const BASE: [ConstModInt<P>; 32] = {
19        let mut base = [ConstModInt::<P>::new(0); 32];
20        let mut t = ConstModInt::<P>::new(P::PRIM_ROOT)
21            ._pow((P::PRIME_NUM as u64 - 1) >> (Self::MAX_POWER));
22
23        let mut i = Self::MAX_POWER;
24        while i > 0 {
25            t = t._mul(t);
26            base[i - 1] = t;
27            i -= 1;
28        }
29
30        base
31    };
32
33    const INV_BASE: [ConstModInt<P>; 32] = {
34        let mut inv_base = [ConstModInt::<P>::new(0); 32];
35        let t = ConstModInt::<P>::new(P::PRIM_ROOT)
36            ._pow((P::PRIME_NUM as u64 - 1) >> (Self::MAX_POWER));
37        let mut s = t._inv();
38
39        let mut i = Self::MAX_POWER;
40        while i > 0 {
41            s = s._mul(s);
42            inv_base[i - 1] = s;
43            i -= 1;
44        }
45
46        inv_base
47    };
48
49    /// [`NTT<P, PRIM_ROOT>`]を作る。
50    pub const fn new() -> Self {
51        Self {
52            _phantom: PhantomData,
53        }
54    }
55
56    /// 数論変換を行う。
57    pub fn ntt(&self, f: &mut [ConstModInt<P>]) {
58        let n = f.len();
59        assert!(n.is_power_of_two() && n < Self::MAX_SIZE);
60
61        let mut b = n / 2;
62        let mut k = n.trailing_zeros() as usize;
63        while b > 0 {
64            let dw = Self::BASE[k];
65
66            let mut ws = vec![ConstModInt::new(1); b];
67            for i in 1..b {
68                ws[i] = ws[i - 1] * dw;
69            }
70
71            for a in f.chunks_exact_mut(2 * b) {
72                let (x, y) = a.split_at_mut(b);
73
74                for ((s, t), &w) in x.iter_mut().zip(y.iter_mut()).zip(ws.iter()) {
75                    let p = *s + *t;
76                    let q = (*s - *t) * w;
77
78                    *s = p;
79                    *t = q;
80                }
81            }
82
83            k -= 1;
84            b >>= 1;
85        }
86
87        // let p = size_of::<usize>() * 8 - n.trailing_zeros() as usize;
88        // let mut g = vec![ConstModInt::new(0); n];
89        // for i in 0..n {
90        //     let j = i.reverse_bits() >> p;
91        //     g[j] = f[i];
92        // }
93        // std::mem::swap(f, &mut g);
94    }
95
96    /// `ntt`の逆変換を行う。
97    pub fn intt(&self, f: &mut [ConstModInt<P>]) {
98        let n = f.len();
99        assert!(n.is_power_of_two() && n < Self::MAX_SIZE);
100
101        // let p = size_of::<usize>() * 8 - n.trailing_zeros() as usize;
102        // let mut g = vec![ConstModInt::new(0); n];
103        // for i in 0..n {
104        //     let j = i.reverse_bits() >> p;
105        //     g[j] = f[i];
106        // }
107        // std::mem::swap(f, &mut g);
108
109        let mut b = 1;
110        let mut k = 1;
111        while b < n {
112            let dw = Self::INV_BASE[k];
113
114            let mut ws = vec![ConstModInt::new(1); b];
115            for i in 1..b {
116                ws[i] = ws[i - 1] * dw;
117            }
118
119            for a in f.chunks_exact_mut(2 * b) {
120                let (x, y) = a.split_at_mut(b);
121
122                for ((s, t), &w) in x.iter_mut().zip(y.iter_mut()).zip(ws.iter()) {
123                    let p = *s + *t * w;
124                    let q = *s - *t * w;
125
126                    *s = p;
127                    *t = q;
128                }
129            }
130
131            k += 1;
132            b <<= 1;
133        }
134
135        let t = ConstModInt::new(n as u32).inv();
136        for x in f.iter_mut() {
137            *x *= t;
138        }
139    }

141    /// 2つの`Vec`を畳み込む。
142    ///
143    /// $(f \ast g)(k) = \sum_{k = i + j} f(i) \times g(j)$
144    pub fn convolve(
145        &self,
146        mut f: Vec<ConstModInt<P>>,
147        mut g: Vec<ConstModInt<P>>,
148    ) -> Vec<ConstModInt<P>> {
149        if f.is_empty() || g.is_empty() {
150            return vec![];
151        }
152
153        let m = f.len() + g.len() - 1;
154        let n = m.next_power_of_two();
155
156        f.resize(n, ConstModInt::new(0));
157        self.ntt(&mut f);
158
159        g.resize(n, ConstModInt::new(0));
160        self.ntt(&mut g);
161
162        for (f, g) in f.iter_mut().zip(g.into_iter()) {
163            *f *= g;
164        }
165        self.intt(&mut f);
166
167        f
168    }
169
170    /// `convolve(f.clone(), f)`と同等。
171    pub fn convolve_same(&self, mut f: Vec<ConstModInt<P>>) -> Vec<ConstModInt<P>> {
172        if f.is_empty() {
173            return vec![];
174        }
175
176        let n = (f.len() * 2 - 1).next_power_of_two();
177        f.resize(n, ConstModInt::new(0));
178
179        self.ntt(&mut f);
180
181        for x in f.iter_mut() {
182            *x *= *x;
183        }
184
185        self.intt(&mut f);
186        f
187    }
188
189    /// NTTで変換可能な配列の最大長を返す。
190    pub fn max_size(&self) -> usize {
191        Self::MAX_SIZE
192    }
193}

204    use super::*;
205    use rand::Rng;

207    #[test]
208    fn test() {
209        type P = Prime<998244353>;
210
211        let ntt = NTT::<P>::new();
212        let ff = ConstModIntBuilder::<P>::new();
213
214        let mut rng = rand::thread_rng();
215
216        let n = rng.gen_range(1..1000);
217        let m = rng.gen_range(1..1000);
218
219        let a = std::iter::repeat_with(|| ff.from_u64(rng.gen_range(0..P::PRIME_NUM) as u64))
220            .take(n)
221            .collect::<Vec<_>>();
222        let b = std::iter::repeat_with(|| ff.from_u64(rng.gen_range(0..P::PRIME_NUM) as u64))
223            .take(m)
224            .collect::<Vec<_>>();
225
226        let res = ntt.convolve(a.clone(), b.clone());
227
228        let mut ans = vec![ConstModInt::new(0); n + m - 1];
229
230        for i in 0..n {
231            for j in 0..m {
232                ans[i + j] += a[i] * b[j];
233            }
234        }
235
236        assert_eq!(&res[..n + m - 1], &ans);
237    }
