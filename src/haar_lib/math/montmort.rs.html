pub fn montmort(n: usize, m: u64) -> Vec<u64> {
    let mut ret = vec![0; n + 1];

    if n >= 2 {
        ret[2] = 1;

        for i in 3..=n {
            ret[i] = (i - 1) as u64 * (ret[i - 1] + ret[i - 2]) % m;
        }
    }

    ret
}

        assert_eq!(
            montmort(20, 998244353)[1..],
            [
                0, 1, 2, 9, 44, 265, 1854, 14833, 133496, 1334961, 14684570, 176214841, 294304226,

    }
