3/// [`crate::io::fastio::FastIO`]を第一引数に、型を第二引数にとって、入力のパースを行う。
4#[macro_export]
5macro_rules! get {
6    ( $in:ident, [$a:tt $(as $to:ty)*; $num:expr] ) => {
7        {
8            let n = $num;
9            (0 .. n).map(|_| get!($in, $a $(as $to)*)).collect::<Vec<_>>()
10        }
11    };
12
13    ( $in:ident, ($($type:tt $(as $to:ty)*),*) ) => {
14        ($(get!($in, $type $(as $to)*)),*)
15    };
16
17    ( $in:ident, i8 ) => { $in.read_i64() as i8 };
18    ( $in:ident, i16 ) => { $in.read_i64() as i16 };
19    ( $in:ident, i32 ) => { $in.read_i64() as i32 };
20    ( $in:ident, i64 ) => { $in.read_i64() };
21    ( $in:ident, isize ) => { $in.read_i64() as isize };
22
23    ( $in:ident, u8 ) => { $in.read_u64() as u8 };
24    ( $in:ident, u16 ) => { $in.read_u64() as u16 };
25    ( $in:ident, u32 ) => { $in.read_u64() as u32 };
26    ( $in:ident, u64 ) => { $in.read_u64() };
27    ( $in:ident, usize ) => { $in.read_u64() as usize };
28
29    ( $in:ident, [char] ) => { $in.read_chars() };
30
31    ( $in:ident, $from:tt as $to:ty ) => { <$to>::from(get!($in, $from)) };
32}

34/// [`crate::io::fastio::FastIO`]を第一引数にとり、第二引数以降に`変数名: 型`を連ねる。
35#[macro_export]
36macro_rules! input {
37    ( @inner $in:ident, mut $name:ident : $type:tt ) => {
38        let mut $name = get!($in, $type);
39    };
40
41    ( @inner $in:ident, mut $name:ident : $type:tt as $to:ty ) => {
42        let mut $name = get!($in, $type as $to);
43    };
44
45    ( @inner $in:ident, $name:ident : $type:tt ) => {
46        let $name = get!($in, $type);
47    };
48
49    ( @inner $in:ident, $name:ident : $type:tt as $to:ty ) => {
50        let $name = get!($in, $type as $to);
51    };
52
53    ( $in:ident >> $($($names:ident)* : $type:tt $(as $to:ty)*),* ) => {
54        $(input!(@inner $in, $($names)* : $type $(as $to)*);)*
55    }
56}

58/// [`crate::io::fastio::FastIO`]を第一引数にとり、第二引数以降を空白区切りで出力する。
59#[macro_export]
60macro_rules! output {
61    ( @one $io:ident, $a:expr ) => {
62        $io.write($a);
63    };
64
65    ( $io:ident << $a:expr, $($rest:expr),* ) => {
66        output!(@one $io, $a);
67        $(
68            $io.write(" ");
69            $io.write($rest);
70        )*
71        $io.writeln("");
72    };
73}
