1//! `impl_algebra!`
2
3#[macro_export]
4macro_rules! impl_algebra {
5    (@bound $t:ty, op: $f:expr; $($bound:tt)*) => {
6        impl $($bound)* BinaryOp for $t {
7            fn op(&self, a: Self::Element, b: Self::Element) -> Self::Element {
8                $f(&self, a, b)
9            }
10        }
11    };
12    (@bound $t:ty, id: $f:expr; $($bound:tt)*) => {
13        impl $($bound)* Identity for $t {
14            fn id(&self) -> Self::Element {
15                $f(&self)
16            }
17        }
18    };
19    (@bound $t:ty, inv: $f:expr; $($bound:tt)*) => {
20        impl $($bound)* Inverse for $t {
21            fn inv(&self, a: Self::Element) -> Self::Element {
22                $f(self, a)
23            }
24        }
25    };
26    (@bound $t:ty, commu: $f:expr; $($bound:tt)*) => {impl $($bound)* Commutative for $t {}};
27    (@bound $t:ty, assoc: $f:expr; $($bound:tt)*) => {impl $($bound)* Associative for $t {}};
28    (@bound $t:ty, idem: $f:expr; $($bound:tt)*) => {impl $($bound)* Idempotence for $t {}};
29    (@bound $t:ty, set: $elem:ty; $($bound:tt)*) => {
30        impl $($bound)* Set for $t {
31            type Element = $elem;
32        }
33    };
34
35    (<const $a:ident : $b:ty>; $t:ty, set: $elem:ty, $($s:ident: $f:expr),+) => {
36        impl_algebra!(@bound $t, set: $elem; <const $a: $b>);
37        $(impl_algebra!(@bound $t, $s: $f; <const $a: $b>);)+
38    };
39    (<$a:ident>; $t:ty, set: $elem:ty, $($s:ident: $f:expr),+) => {
40        impl_algebra!(@bound $t, set: $elem; <$a>);
41        $(impl_algebra!(@bound $t, $s: $f; <$a>);)+
42    };
43    ($t:ty, set: $elem:ty, $($s:ident: $f:expr),+) => {
44        impl_algebra!(@bound $t, set: $elem;);
45        $(impl_algebra!(@bound $t, $s: $f;);)+
46    };
47}
