1//! 代数的構造に関するトレイトを定義する。
2use crate::trait_alias;
3
4/// 集合
5pub trait Set {
6    /// 集合の要素の型
7    type Element;
8}
9
10/// 二項演算をもつ
11pub trait BinaryOp: Set {
12    /// 二項演算
13    fn op(&self, _: Self::Element, _: Self::Element) -> Self::Element;
14}
15
16/// 単位元をもつ
17pub trait Identity: Set {
18    /// 単位元
19    fn id(&self) -> Self::Element;
20}
21
22/// 逆元をもつ
23pub trait Inverse: Set {
24    /// 逆元
25    fn inv(&self, _: Self::Element) -> Self::Element;
26}
27
28/// 可換性をもつ
29pub trait Commutative {}
30/// 結合性をもつ
31pub trait Associative {}
32/// 冪等性をもつ
33pub trait Idempotence {}
34
35trait_alias!(#[doc = "半群"] Semigroup: BinaryOp + Associative);
36trait_alias!(#[doc = "モノイド"] Monoid: Semigroup + Identity);
37trait_alias!(#[doc = "可換モノイド"] AbelianMonoid: Monoid + Commutative);
38trait_alias!(#[doc = "群"] Group: Monoid + Inverse);
39trait_alias!(#[doc = "可換群"] AbelianGroup: Group + Commutative);

41/// 値に二項演算を複数回適用する。
42pub trait Times: BinaryOp + Identity
43where
44    Self::Element: Clone,
45{
46    /// `n`個の値`a`に二項演算を適用する。
47    fn times(&self, mut a: Self::Element, mut n: u64) -> Self::Element {
48        let mut ret = self.id();
49
50        while n > 0 {
51            if n & 1 == 1 {
52                ret = self.op(ret, a.clone());
53            }
54            a = self.op(a.clone(), a);
55            n >>= 1;
56        }
57
58        ret
59    }
60}
61impl<A: BinaryOp + Identity> Times for A where Self::Element: Clone {}
